//virtual function and runtime polymorphism
#include<iostream>
using namespace std;
class Animal
{
	public:
		virtual void sound(){
			cout<<"Animal makes a sound"<<endl;
		}
};
class Dog:public Animal{
	public:
		void sound() override{
		cout<<"Dog barks"<<endl;
		}
};
class Cat:public Animal{
	public:
		void sound() override{
		cout<<"Cat meow"<<endl;
		}
};
int main()
{
	Animal *a;
	Dog d;
	Cat c;
	Animal base;
	a=&d;
	a->sound();
	a=&c;
	a->sound();
	a=&base;
	a->sound();
	return 0;
}
